local TeleStep = 0.3
local TeleRayDist = 3

local function setupTeleCharacter(char)
    TeleCharacter = char

    TeleHum = char:WaitForChild("Humanoid")
end

RunService.Heartbeat:Connect(function()
    if not Cfg.TeleSpeed then return end
    if not TeleRoot or not TeleHum or TeleHum.Health <= 0 then return end

    local moveKeyDown = false
    for _, key in ipairs(TeleMovementKeys) do
        if UIS:IsKeyDown(key) then moveKeyDown = true; break end
    end
    if not moveKeyDown then return end

    local moveDir = TeleHum.MoveDirection
    if moveDir.Magnitude > 0.1 then
        local rayOrigin = TeleRoot.Position
        local rayDirection = moveDir * TeleRayDist
        local hit = Workspace:Raycast(rayOrigin, rayDirection)

        local distance = TeleStep
        if hit then
            distance = math.min(TeleStep, (hit.Position - rayOrigin).Magnitude - 1.5)
            if distance < 0 then distance = 0 end
        end

        if distance > 0 then
            TeleRoot.CFrame = TeleRoot.CFrame + moveDir * distance
        end
    end
end)

function ESPIndicator.new(o)
local s = setmetatable({}, ESPIndicator)
s.Settings = {}
for k, v in pairs(ESPIndicator.Defaults) do
s.Settings[k] = (o and o[k] ~= nil) and o[k] or v
end
local p = s.Settings.Parent or Player:WaitForChild("PlayerGui")
s.ScreenGui = Instance.new("ScreenGui")
s.ScreenGui.Name = "ESPIndicators"
s.ScreenGui.IgnoreGuiInset = true
s.ScreenGui.ResetOnSpawn = false
s.ScreenGui.Parent = p
s.Indicators = {}
return s
end
function ESPIndicator:AddGroup(n)
local g = ESPIndicator.Groups[n]
if not g then g = {enabled = true, properties = {}, targets = {}}; ESPIndicator.Groups[n] = g end
return g
end
function ESPIndicator:RemoveGroup(n)
local g = ESPIndicator.Groups[n]
if not g then return false end
for _, t in ipairs(g.targets) do
local idx = ESPIndicator.TargetIndex[t]
if idx then
for i, nm in ipairs(idx) do if nm == n then table.remove(idx, i); break end end
if #idx == 0 then ESPIndicator.TargetIndex[t] = nil end
end
if not ESPIndicator.TargetIndex[t] then self:Remove(t) end
end
ESPIndicator.Groups[n] = nil
return true
end
function ESPIndicator:ClearAllGroups()
for n in pairs(ESPIndicator.Groups) do self:RemoveGroup(n) end
end
function ESPIndicator:Add(target, opts)
assert(target)
opts = opts or {}
local hl = Instance.new("Highlight")
hl.Name = "HL_" .. HttpService:GenerateGUID(false)
hl.Adornee = target
hl.FillTransparency = opts.HighlightFillTransparency or self.Settings.HighlightFillTransparency
hl.FillColor = opts.AccentColor or self.Settings.AccentColor
hl.OutlineColor = opts.AccentColor or self.Settings.AccentColor
hl.OutlineTransparency = opts.HighlightOutlineTransparency or self.Settings.HighlightOutlineTransparency
hl.DepthMode = opts.HighlightDepthMode or self.Settings.HighlightDepthMode
hl.Parent = self.ScreenGui
self.Indicators[target] = { Highlight = hl, Options = opts }
local gn = opts.GroupName
if gn then self:AddToGroup(target, gn) end
end
function ESPIndicator:Remove(t)
local ind = self.Indicators[t]
if not ind then return end
if ind.Highlight then ind.Highlight.Adornee = nil; ind.Highlight:Destroy() end
if ind.HPGui then
pcall(function() ind.HPGui:Destroy() end)
end
local idx = ESPIndicator.TargetIndex[t]
if idx then
for _, gn in ipairs(idx) do
local g = ESPIndicator.Groups[gn]
if g then
for i, tt in ipairs(g.targets) do if tt == t then table.remove(g.targets, i); break end end
end
end
ESPIndicator.TargetIndex[t] = nil
end
self.Indicators[t] = nil
end
function ESPIndicator:AddToGroup(t, gn)
local g = self:AddGroup(gn)
if not table.find(g.targets, t) then table.insert(g.targets, t) end
local idx = ESPIndicator.TargetIndex[t]
if not idx then idx = {}; ESPIndicator.TargetIndex[t] = idx end
if not table.find(idx, gn) then table.insert(idx, gn) end
return true
end
function ESPIndicator:Destroy()
self:ClearAllGroups()
for _, ind in pairs(self.Indicators) do
if ind.Highlight then ind.Highlight:Destroy() end
if ind.HPGui then pcall(function() ind.HPGui:Destroy() end) end
end
self.ScreenGui:Destroy()
self.Indicators = {}
end

ItemESPObjects[model] = nil
end
end
local function updateItemESP()
local char = Player.Character
local hrp = char and char:FindFirstChild("HumanoidRootPart")
for model, e in pairs(ItemESPObjects) do
if not model or not model.Parent then
removeItemESP(model)
elseif hrp then
local ok, dist = pcall(function()
local p = e.root
if p and p:IsA("BasePart") then
return math.floor((hrp.Position - p.Position).Magnitude)
end
return 999
end)
if ok and dist then
if dist > ESPSettings.MaxDistance then
e.hl.Enabled = false
e.bb.Enabled = false
else
e.hl.Enabled = true
e.bb.Enabled = true
e.lbl.Text = model.Name .. "\n" .. dist .. "m"
end
end
end
end
end

local function updateGenCache()
local now = tick()
if now - lastGenScan < 2 then return end
lastGenScan = now
table.clear(genCache)
local ingame = Workspace:FindFirstChild("Map") and Workspace.Map:FindFirstChild("Ingame")
if not ingame then return end
local map = ingame:FindFirstChild("Map")
if not map then return end
for _, gen in ipairs(map:GetChildren()) do
if gen.Name == "Generator" and gen:IsA("Model") and not gen:GetFullName():find("Lobby") then
local main = gen:FindFirstChild("Main")
if main then table.insert(genCache, { Model=gen, Main=main }) end
end
end
end
local function getNearestGenerator()
updateGenCache()
local hrp = getHRP()
if not hrp then return nil, 999 end
local nearest, nearestDist = nil, math.huge
for _, data in ipairs(genCache) do
local gen, main = data.Model, data.Main
local prog = gen:FindFirstChild("Progress") or gen:FindFirstChildWhichIsA("NumberValue")
if prog and prog.Value >= 100 then continue end
local d = (hrp.Position - main.Position).Magnitude
if d < nearestDist then nearestDist = d; nearest = gen end
end
return nearest, nearestDist
end
local function fireNearest()
local hrp = getHRP()
if not hrp then return end
local gen, d = getNearestGenerator()
if not gen or d > 15 then return end
local now = tick()
local last = lastFired[gen] or 0
if now - last < GEN_COOLDOWN then return end
local rem = gen:FindFirstChild("Remotes")
if not rem then return end
local re = rem:FindFirstChild("RE")
if not re then return end
lastFired[gen] = now
pcall(function() re:FireServer() end)
end

local AntiDebuff = {Enabled = true}

local char = Player.Character

local hum = char:FindFirstChild("Humanoid")
if hum and hum.WalkSpeed < 16 then hum.WalkSpeed = 16 end

local nameLower = child.Name:lower()

end)
end
