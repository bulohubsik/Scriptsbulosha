local function httpRequest(url, method, headers, body)
    method  = method  or "GET"
    headers = headers or {}
    body    = body    or nil

    if type(request) == "function" then
        local ok, res = pcall(request, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if type(syn) == "table" and type(syn.request) == "function" then
        local ok, res = pcall(syn.request, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if type(http_request) == "function" then
        local ok, res = pcall(http_request, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if type(fluxus_request) == "function" then
        local ok, res = pcall(fluxus_request, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if type(krnl_request) == "function" then
        local ok, res = pcall(krnl_request, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if type(HttpRequest) == "function" then
        local ok, res = pcall(HttpRequest, {Url=url, Method=method, Headers=headers, Body=body})
        if ok and res then return res end
    end
    if method == "GET" then
        local ok, res = pcall(function() return game:HttpGet(url) end)
        if ok and res then return {Body=res, StatusCode=200} end
    end
    if method == "GET" then
        local ok, res = pcall(function() return HttpService:GetAsync(url) end)
        if ok and res then return {Body=res, StatusCode=200} end
    end
    if method == "POST" and body then
        local ok, res = pcall(function()
            return HttpService:PostAsync(url, body, Enum.HttpContentType.ApplicationJson)
        end)
        if ok then return {Body=res or "", StatusCode=200} end
    end
    local globalFuncs = {"fetch","http","httpGet","httpPost","doRequest","sendRequest"}
    for _, fname in ipairs(globalFuncs) do
        if type(_G[fname]) == "function" then
            local ok, res = pcall(_G[fname], {Url=url, Method=method, Headers=headers, Body=body})
            if ok and res then return res end
        end
    end
    return {Body="", StatusCode=0}
end

local function safeHttpGet(url)
    local ok, res = pcall(function() return game:HttpGet(url) end)
    if ok and res and res ~= "" then return res end
    local ok2, res2 = pcall(function() return game:HttpGet(url, true) end)
    if ok2 and res2 and res2 ~= "" then return res2 end
    return nil
end

local function safeLoadstring(src)
    if not src then return nil, "nil source" end
    if loadstring then
        local fn, err = loadstring(src)
        return fn, err
    end
    return nil, "loadstring unavailable"
end

local function getDeviceInfo()
    local ok, UIS2 = pcall(function() return game:GetService("UserInputService") end)
    if ok and UIS2 then
        local touch    = pcall(function() return UIS2.TouchEnabled    end) and UIS2.TouchEnabled    or false
        local keyboard = pcall(function() return UIS2.KeyboardEnabled end) and UIS2.KeyboardEnabled or false
        local gamepad  = pcall(function() return UIS2.GamepadEnabled  end) and UIS2.GamepadEnabled  or false
        if touch    and not keyboard then return "Mobile"  end
        if gamepad  and not keyboard then return "Console" end
        if keyboard                  then return "PC"      end
    end
    return "Unknown"
end

    end

    local checks = {

local function sendWebhook(url, title, fields, color)
    pcall(function()
        local payload = HttpService:JSONEncode({
            embeds = {{
                title  = title,
                color  = color or 0x0066FF,
                fields = fields,
                footer = {text = "v" .. HUB_VERSION .. " • " .. HUB_NAME}
            }},
            username = HUB_NAME,
        })
        httpRequest(url, "POST", {["Content-Type"]="application/json"}, payload)
    end)
end

local _drawingEnabled = (Drawing ~= nil)

local function newDrawing(class, props)
    if not _drawingEnabled then
        local dummy = {Visible=false, Remove=function() end}
        return setmetatable(dummy, {
            __index    = function(_, k) return rawget(dummy, k) end,
            __newindex = function(t, k, v) rawset(t, k, v) end,
        })
    end
    local ok, d = pcall(function() return Drawing.new(class) end)
    if not ok then
        local dummy = {Visible=false, Remove=function() end}
        return setmetatable(dummy, {__newindex=function(t,k,v) rawset(t,k,v) end})
    end
    if props then for k, v in pairs(props) do pcall(function() d[k]=v end) end end
    return d
end

local function applyAntiSpread()
    if _antiSpreadPatched then return end
    for _, v in pairs(getgc(true)) do
        if type(v)=="table" and rawget(v,"getTrueSpread") and rawget(v,"_performRaycast") then
            _origGetTrueSpread = v.getTrueSpread
            v.getTrueSpread    = function() return 0 end
            _antiSpreadPatched = true
            break
        end
    end
end

local function removeAntiSpread()
    if not _antiSpreadPatched then return end
    for _, v in pairs(getgc(true)) do
        if type(v)=="table" and rawget(v,"getTrueSpread") and rawget(v,"_performRaycast") then
            if _origGetTrueSpread then v.getTrueSpread = _origGetTrueSpread end
            break
        end
    end
    _antiSpreadPatched = false
    _origGetTrueSpread = nil
end

local function applySpeedBoost()
    if _speedBoostPatched then return end
    for _, v in pairs(getgc(true)) do
        if type(v)=="table" and rawget(v,"GetMaxSpeed") then

end

local _visRP = RaycastParams.new()
_visRP.FilterType = Enum.RaycastFilterType.Exclude

local function canSee(part)
    local ignore = {}
    if player.Character then table.insert(ignore, player.Character) end
    if part.Parent      then table.insert(ignore, part.Parent)      end
    _visRP.FilterDescendantsInstances = ignore
    local origin = camera.CFrame.Position
    local result = Workspace:Raycast(origin, part.Position-origin, _visRP)
    return result == nil
end

local function UpdateAim()
    if _drawingEnabled then
        if not _aimbot_circle then
            _aimbot_circle = newDrawing("Circle", {
                Filled=false, Thickness=1,
                Color=Color3.fromRGB(255,255,255), Transparency=0.6,
            })
        end
        local mp = UIS:GetMouseLocation()
        _aimbot_circle.Position = mp
        _aimbot_circle.Radius   = _aim.fov
        _aimbot_circle.Visible  = _aim.show_fov
    end

    if not _aim.enabled then return end
    if not _aim.always_on and not _aimbot_active then return end
    if not isAlive() then return end

    local ef         = getEnemyFolder()
    local local_char = player.Character
    local local_pos  = local_char and local_char:GetPivot().Position
    local mouse_pos  = UIS:GetMouseLocation()
    local best_target, best_score = nil, math.huge

    if ef then
        for _, e in ipairs(ef:GetChildren()) do
            local humanoid = e:FindFirstChildOfClass("Humanoid")
            if not humanoid or humanoid.Health <= 0 then continue end
            local part = e:FindFirstChild(_aim.aim_part) or e:FindFirstChild("Head")
            if not part or not part:IsA("BasePart") then continue end
            if not canSee(part) then continue end
            local sp    = camera:WorldToViewportPoint(part.Position)
            local sd    = (Vector2.new(sp.X,sp.Y)-mouse_pos).Magnitude
            local wd    = local_pos and (local_pos-e:GetPivot().Position).Magnitude or 0
            local score = sd + (wd*_aim.distance_bias)
            if sd <= _aim.fov and score < best_score then
                best_score  = score
                best_target = part
            end
        end
    end

    if best_target then
        local ts      = camera:WorldToViewportPoint(best_target.Position)
        local divisor = math.max(_aim.smoothing*10, 1)
        local dx      = (ts.X-mouse_pos.X)/divisor
        local dy      = (ts.Y-mouse_pos.Y)/divisor
        local moved   = false
        if _mousemoverel then
            local ok = pcall(_mousemoverel, dx, dy)
            if ok then moved = true end
        end
        if not moved then
            local alpha = 1 - math.clamp(_aim.smoothing, 0, 1)
            if alpha > 0 then
                pcall(function()
                    camera.CFrame = camera.CFrame:Lerp(
                        CFrame.new(camera.CFrame.Position, best_target.Position), alpha
                    )
                end)
            end
        end
    end
end

        local res = Workspace:Raycast(ray.Origin, ray.Direction*1000, rp)
        if res then

            if ef and model and model.Parent==ef then
                local hum = model:FindFirstChildOfClass("Humanoid")

                    if TBotDelay>0 then task.wait(TBotDelay/1000) end
                    pcall(_mouse1click)
                    task.wait(0.05)
                end
            end
        end
    end

local function cleanupLV()
    if speedLV  then pcall(function() speedLV:Destroy()  end); speedLV  = nil end
    if speedAtt then pcall(function() speedAtt:Destroy() end); speedAtt = nil end
end

local function getOrCreateLV()
    local char = player.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then cleanupLV(); return nil end
    if speedLV and speedLV.Parent==hrp then return speedLV end
    cleanupLV()
    local ok, result = pcall(function()
        local att = Instance.new("Attachment")
        att.Name="SpeedAtt"; att.Parent=hrp; speedAtt=att
        local lv = Instance.new("LinearVelocity")
        lv.Name="LV_Speed"; lv.Attachment0=att; lv.MaxForce=8000
        lv.RelativeTo=Enum.ActuatorRelativeTo.World
        lv.VelocityConstraintMode=Enum.VelocityConstraintMode.Vector
        lv.VectorVelocity=Vector3.zero; lv.Parent=hrp; speedLV=lv
        return lv
    end)
    if not ok then return nil end
    return result
end

local _keyDirs = {
    [Enum.KeyCode.W] = Vector3.new( 0,0,-1),
    [Enum.KeyCode.S] = Vector3.new( 0,0, 1),
    [Enum.KeyCode.A] = Vector3.new(-1,0, 0),
    [Enum.KeyCode.D] = Vector3.new( 1,0, 0),
}

local function getInputDir()
    local dir = Vector3.zero
    for key, vec in pairs(_keyDirs) do
        if UIS:IsKeyDown(key) then dir=dir+vec end
    end
    if dir.Magnitude==0 then return Vector3.zero end
    local look = camera.CFrame.LookVector
    local flat  = Vector3.new(look.X,0,look.Z)
    if flat.Magnitude<0.01 then return Vector3.zero end
    return (CFrame.new(Vector3.zero,flat):VectorToWorldSpace(dir)).Unit
end

local function cleanPart(p)
    if not p:IsA("BasePart") then return end
    pcall(function()
        p.CanCollide=false; p.Anchored=false
        p.CastShadow=false; p.CanTouch=false; p.CanQuery=false
    end)
end

local function disableAll(m)
    for _, p in ipairs(m:GetDescendants()) do cleanPart(p) end
end

local function hideKnife(k)
    if not k then return end
    pcall(function()
        for _, p in ipairs(k:GetDescendants()) do
            if p:IsA("BasePart") or p:IsA("MeshPart") or p:IsA("Texture") then
                p.Transparency=1
            end
        end
    end)
end

local function kSound(folder, name)
    local ok, result = pcall(function()
        local ws = RS.Sounds:FindFirstChild(selectedKnife)
        if not ws then return nil end
        local s = ws:WaitForChild(folder):WaitForChild(name):Clone()
        s.Parent=camera; s:Play()
        s.Ended:Once(function() s:Destroy() end)
        return s
    end)
    if ok then return result end
end

local function attachAsset(folder, armN, assetN, finalN, off)
    pcall(function()
        local arm = vm:FindFirstChild(armN)
        if not arm then return end
        local mesh = folder:WaitForChild(assetN):Clone()
        cleanPart(mesh); mesh.Name=finalN; mesh.Parent=arm
        local m6 = Instance.new("Motor6D")
        m6.Part0=arm; m6.Part1=mesh; m6.C0=off; m6.Parent=arm
    end)
end

local function handleKnifeAction(action, state)
    if state~=Enum.UserInputState.Begin then return Enum.ContextActionResult.Pass end
    if not spawned or not animator or not isAlive() then return Enum.ContextActionResult.Pass end
    if action==ACT_INS then
        if (eqAnim and eqAnim.IsPlaying) or inspecting or swinging then return Enum.ContextActionResult.Pass end
        inspecting=true
        if idAnim then pcall(function() idAnim:Stop() end) end
        pcall(function() insAnim:Play() end)
        insAnim.Stopped:Once(function() inspecting=false end)
    elseif action==ACT_ATK then
        local now=os.clock()
        if (eqAnim and eqAnim.IsPlaying) or now-lastAttack<KNIFE_CD then return Enum.ContextActionResult.Pass end
        lastAttack=now
        if inspecting then
            inspecting=false
            if insAnim then pcall(function() insAnim:Stop() end) end
        end
        swinging=true
        if idAnim then pcall(function() idAnim:Stop() end) end
        local pool={hvAnim,sw1Anim,sw2Anim}
        local ch=pool[math.random(1,#pool)]
        local sfx=ch==hvAnim and "HitOne" or ch==sw1Anim and "HitTwo" or "HitThree"
        pcall(function() ch:Play() end)
        local s=kSound(sfx,"1")
        if s then pcall(function() s.Volume=5 end) end
        ch.Stopped:Once(function() swinging=false end)
    end
    return Enum.ContextActionResult.Pass
end

local function removeVM()
    spawned=false; inspecting=false; swinging=false
    pcall(function() CAS:UnbindAction(ACT_INS) end)
    pcall(function() CAS:UnbindAction(ACT_ATK) end)
    if vm then pcall(function() vm:Destroy() end); vm=nil end
    animator=nil
end

local function spawnVM(knife)
    if spawned or not scriptRunning then return end
    local myTeam=getMyTeamAndEnemy()
    if not myTeam then return end
    spawned=true
    pcall(function()
        local tmpl  = RS.Assets.Weapons:WaitForChild(selectedKnife)
        local offset= knivesData[selectedKnife].Offset
        vm          = tmpl:WaitForChild("Camera"):Clone()
        vm.Name=selectedKnife; vm.Parent=camera
        disableAll(vm); hideKnife(knife)
        if myTeam.Name=="Terrorists" then
            local tg=RS.Assets.Weapons:WaitForChild("T Glove")
            attachAsset(tg,"Left Arm","Left Arm","Glove",CFrame.new(0,0,-1.5))
            attachAsset(tg,"Right Arm","Right Arm","Glove",CFrame.new(0,0,-1.5))
        else
            local sl=RS.Assets.Sleeves:WaitForChild("IDF")
            local cg=RS.Assets.Weapons:WaitForChild("CT Glove")
            attachAsset(sl,"Left Arm","Left Arm","Sleeve",CFrame.new(0,0,0.5))
            attachAsset(cg,"Left Arm","Left Arm","Glove",CFrame.new(0,0,-1.5))
            attachAsset(sl,"Right Arm","Right Arm","Sleeve",CFrame.new(0,0,0.5))
            attachAsset(cg,"Right Arm","Right Arm","Glove",CFrame.new(0,0,-1.5))
        end
        local ac=vm:FindFirstChildOfClass("AnimationController") or vm:FindFirstChildOfClass("Animator")
        animator=ac:FindFirstChildWhichIsA("Animator") or ac
        local af=RS.Assets.WeaponAnimations:WaitForChild(selectedKnife):WaitForChild("CameraAnimations")
        eqAnim  = animator:LoadAnimation(af:WaitForChild("Equip"))
        idAnim  = animator:LoadAnimation(af:WaitForChild("Idle"))
        insAnim = animator:LoadAnimation(af:WaitForChild("Inspect"))
        hvAnim  = animator:LoadAnimation(af:WaitForChild("Heavy Swing"))
        sw1Anim = animator:LoadAnimation(af:WaitForChild("Swing1"))
        sw2Anim = animator:LoadAnimation(af:WaitForChild("Swing2"))
        vm:SetPrimaryPartCFrame(camera.CFrame*CFrame.new(0,-1.5,5))
        TweenService:Create(vm.PrimaryPart,TweenInfo.new(0.2,Enum.EasingStyle.Quad,Enum.EasingDirection.Out),
            {CFrame=camera.CFrame*offset}):Play()
        eqAnim:Play(); kSound("Equip","1")
        CAS:BindAction(ACT_INS,handleKnifeAction,false,Enum.KeyCode.F)
        CAS:BindAction(ACT_ATK,handleKnifeAction,false,Enum.UserInputType.MouseButton1)
    end)
end

local function applyWeaponSkin(model)
    if not model or not SkinChangerEnabled or not isAlive() then return end
    local sn=SelectedSkins[model.Name]; if not sn then return end
    pcall(function()
        local sf=SkinsFolder:FindFirstChild(model.Name); if not sf then return end
        local st=sf:FindFirstChild(sn); if not st then return end
        local src=st:FindFirstChild("Camera") and st.Camera:FindFirstChild(WEAR)
        if not src then return end
        for _, obj in ipairs(camera:GetChildren()) do
            if obj:FindFirstChild("Left Arm") or obj:FindFirstChild("Right Arm") then
                local gf=SkinsFolder:FindFirstChild("Sports Gloves")
                local gs=gf and gf:FindFirstChild(SelectedSkins["Sports Gloves"])
                local gsrc=gs and gs:FindFirstChild("Camera") and gs.Camera:FindFirstChild(WEAR)
                if gsrc then
                    for _, side in ipairs({"Left Arm","Right Arm"}) do
                        local arm=obj:FindFirstChild(side)
                        local asrc=gsrc:FindFirstChild(side)
                        if arm and asrc then
                            local gm=arm:FindFirstChild("Glove")
                            if gm then
                                local ex=gm:FindFirstChildOfClass("SurfaceAppearance")
                                if ex then ex:Destroy() end
                                local cl=asrc:Clone(); cl.Name="SurfaceAppearance"; cl.Parent=gm
                            end
                        end
                    end
                end
            end
        end
        if not GLOVES[model.Name] then
            local wf=model:FindFirstChild("Weapon")
            if wf then
                for _, part in ipairs(wf:GetDescendants()) do
                    if part:IsA("BasePart") then
                        local ns=src:FindFirstChild(part.Name)
                        if ns then
                            local ex=part:FindFirstChildOfClass("SurfaceAppearance")
                            if ex then ex:Destroy() end
                            local cl=ns:Clone(); cl.Name="SurfaceAppearance"; cl.Parent=part
                        end
                    end
                end
            end
        end
        model:SetAttribute("SkinApplied",sn)
    end)
end

                if hum then
                    if lastHp[e] and lastHp[e]>0 and hum.Health<=0 then doKillFX() end
                    lastHp[e]=hum.Health
                end
            end
        end)
    end
end)

RunService.RenderStepped:Connect(function(dt)
    pcall(UpdateAim)

    local char = player.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    local hum  = char and char:FindFirstChildOfClass("Humanoid")

    pcall(function()
        if BhopEnabled and isAlive() and hum and hrp then
            if UIS:IsKeyDown(Enum.KeyCode.Space) then
                local st     = hum:GetState()
                local curVel = hrp.AssemblyLinearVelocity
                if st~=Enum.HumanoidStateType.Jumping
                and st~=Enum.HumanoidStateType.Freefall
                and curVel.Y<=1 then
                    hum.Jump=true
                end
            end
        end
    end)

    pcall(function()
        if SpeedEnabled and isAlive() and hrp and hum then
            local lv       = getOrCreateLV()
            local dir      = getInputDir()
            local curVel   = hrp.AssemblyLinearVelocity
            local curY     = curVel.Y
            local humState = hum:GetState()
            local inAir    = humState==Enum.HumanoidStateType.Freefall or humState==Enum.HumanoidStateType.Jumping
            local rawSpd   = SpeedValue*_jitterValue
            local horizSpd = inAir and math.min(rawSpd,HORIZ_MAX) or rawSpd
            if lv then
                if dir.Magnitude>0 then
                    lv.VectorVelocity=Vector3.new(dir.X*horizSpd,curY,dir.Z*horizSpd)
                    _lastHorizVel=Vector3.new(dir.X*horizSpd,0,dir.Z*horizSpd)
                else
                    local brake=inAir and 0.93 or 0.7
                    _lastHorizVel=_lastHorizVel*brake
                    if _lastHorizVel.Magnitude<0.5 then _lastHorizVel=Vector3.zero end
                    lv.VectorVelocity=Vector3.new(_lastHorizVel.X,curY,_lastHorizVel.Z)
                end
            end
            if JumpBoostOn then
                local justJumped=humState==Enum.HumanoidStateType.Jumping
                if justJumped and not _jumpedLastFrame then
                    local safeY=math.min(math.max(curY,0)+JumpBoostPower,JUMP_Y_MAX)
                    pcall(function() hrp.AssemblyLinearVelocity=Vector3.new(curVel.X,safeY,curVel.Z) end)
                end
                if curVel.Y>JUMP_Y_MAX then
                    pcall(function() hrp.AssemblyLinearVelocity=Vector3.new(curVel.X,JUMP_Y_MAX,curVel.Z) end)
                end
                _jumpedLastFrame=justJumped
            else
                _jumpedLastFrame=false
            end
        elseif not SpeedEnabled then
            if speedLV then
                pcall(function() speedLV.VectorVelocity=Vector3.zero end)
                task.delay(0.05,cleanupLV)
            end
            _lastHorizVel=Vector3.zero; _jumpedLastFrame=false
        end
    end)

    pcall(function()
        if scriptRunning and vm and vm.PrimaryPart then
            vm.PrimaryPart.CFrame=camera.CFrame*knivesData[selectedKnife].Offset
            if not (eqAnim and eqAnim.IsPlaying) and not inspecting and not swinging then
                if idAnim and not idAnim.IsPlaying then idAnim:Play() end
            end
        end
    end)

    pcall(function()
        local ef=getEnemyFolder()
        if not ESP.Enabled or not isAlive() or not ef then
            for _, obj in pairs(ESPObjects) do hideESPObject(obj) end
            return
        end
        for enemy, obj in pairs(ESPObjects) do
            if not enemy.Parent or enemy.Parent~=ef then hideESPObject(obj) end
        end
        for _, enemy in ipairs(ef:GetChildren()) do
            local ehum  = enemy:FindFirstChildOfClass("Humanoid")
            local ehrp  = enemy:FindFirstChild("HumanoidRootPart")
            local ehead = enemy:FindFirstChild("Head")
            if not ehum or not ehrp or not ehead or ehum.Health<=0 then
                if ESPObjects[enemy] then hideESPObject(ESPObjects[enemy]) end; continue
            end
            if not ESPObjects[enemy] then ESPObjects[enemy]=createESPObject() end
            local obj  = ESPObjects[enemy]
            local dist = (camera.CFrame.Position-ehrp.Position).Magnitude
            if dist>ESP.MaxDistance then hideESPObject(obj); continue end
            local sHead,onScreen = w2s(ehead.Position+Vector3.new(0,ehead.Size.Y/2+0.1,0))
            local sFeet          = w2s(ehrp.Position-Vector3.new(0,3,0))
            local sRoot          = w2s(ehrp.Position)
            if not onScreen then hideESPObject(obj); continue end
            local bH=math.max(12,math.abs(sFeet.Y-sHead.Y)*1.05)
            local bW=math.max(6, bH*0.45)
            local bX=sRoot.X-bW/2; local bY=sHead.Y-2
            local hp=ehum.Health; local mhp=ehum.MaxHealth
            local pct=math.clamp(hp/mhp,0,1)
            pcall(function()
                obj.BoxOL.Position=Vector2.new(bX-1,bY-1); obj.BoxOL.Size=Vector2.new(bW+2,bH+2)
                obj.BoxOL.Thickness=ESP.BoxThickness+1;    obj.BoxOL.Visible=true
                obj.Box.Position=Vector2.new(bX,bY);       obj.Box.Size=Vector2.new(bW,bH)
                obj.Box.Color=ESP.BoxColor; obj.Box.Thickness=ESP.BoxThickness; obj.Box.Visible=true
                if ESP.ShowName then
                    obj.Name.Text=enemy.Name; obj.Name.Size=ESP.TextSize; obj.Name.Color=ESP.TextColor
                    obj.Name.Position=Vector2.new(sRoot.X,bY-ESP.TextSize-2); obj.Name.Visible=true
                else obj.Name.Visible=false end
                if ESP.ShowDistance then
                    obj.Dist.Text=string.format("[%dm]",math.floor(dist/3.5))
                    obj.Dist.Size=math.max(8,ESP.TextSize-2)
                    obj.Dist.Position=Vector2.new(sRoot.X,bY+bH+3); obj.Dist.Visible=true
                else obj.Dist.Visible=false end
                if ESP.ShowHealthBar then
                    local bw=ESP.HealthBarWidth; local barX=bX-bw-3
                    local fillH=math.max(1,bH*pct); local hpCol=GetHealthColor(hp,mhp)
                    obj.HpBg.Position=Vector2.new(barX,bY);     obj.HpBg.Size=Vector2.new(bw,bH);     obj.HpBg.Visible=true
                    obj.HpOL.Position=Vector2.new(barX-1,bY-1); obj.HpOL.Size=Vector2.new(bw+2,bH+2); obj.HpOL.Visible=true
                    obj.HpBar.Position=Vector2.new(barX,bY+bH-fillH); obj.HpBar.Size=Vector2.new(bw,fillH)
                    obj.HpBar.Color=hpCol; obj.HpBar.Visible=true
                    if pct<1 then
                        obj.HpTxt.Text=tostring(math.floor(hp))
                        obj.HpTxt.Position=Vector2.new(barX+bw/2,bY+bH-fillH-11)
                        obj.HpTxt.Color=hpCol; obj.HpTxt.Visible=true
                    else obj.HpTxt.Visible=false end
                else
                    obj.HpBg.Visible=false; obj.HpOL.Visible=false
                    obj.HpBar.Visible=false; obj.HpTxt.Visible=false
                end
            end)
        end
    end)
end)
